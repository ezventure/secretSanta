<body>


<script>

    // Secret Santa participants

    const participants = [

        { name: "Mom", id: "Mom" },

        { name: "Dad", id: "Dad" },

        { name: "Zoya", id: "Zoya" },

        { name: "Devina", id: "Devina" },

        { name: "Alden", id: "Alden" },

        { name: "Dwi", id: "Dwi" },

        { name: "Yehez", id: "Yehez" },

    ];


    // Function to shuffle an array

    function shuffleArray(array) {

        for (let i = array.length - 1; i > 0; i--) {

            const j = Math.floor(Math.random() * (i + 1));

            [array[i], array[j]] = [array[j], array[i]];

        }

    }


    // Function to create Secret Santa pairings

    function createSecretSantaPairings(participant) {

        const pairings = [];

        const shuffledParticipants = [...participants];

        shuffleArray(shuffledParticipants);


        for (let i = 0; i < participants.length; i++) {

            const giver = participant.name;

            const recipient1 = shuffledParticipants[i % participants.length].name;

            const recipient2 = shuffledParticipants[(i + 1) % participants.length].name;


            pairings.push(`${giver} gives a gift to ${recipient1} and ${recipient2}`);

        }


        return pairings;

    }


    // Function to update the recipient names in the text box

    function updateRecipientNames(pairings) {

        const resultTextBox = document.getElementById('resultTextBox');

        resultTextBox.textContent = pairings.join('\n');

    }


    // Event handler for the "Assemble" button

    function assemble() {

        const participantDropdown = document.getElementById('participantDropdown');

        const selectedParticipantId = participantDropdown.value;


        const participant = participants.find(p => p.id === selectedParticipantId);


        if (participant) {

            const secretSantaPairings = createSecretSantaPairings(participant);

            updateRecipientNames(secretSantaPairings);

        } else {

            console.error('Participant not found.');

        }

    }

</script>


<!-- HTML content -->

<h1>Secret Santa</h1>


<!-- Participant dropdown -->

<label for="participantDropdown">Select Participant:</label>

<select id="participantDropdown">

    <option value="Mom">Mom</option>

    <option value="Dwi">Dwi</option>

    <option value="Yehez">Yehez</option>

</select>


<!-- Assemble button -->

<button onclick="assemble()">Assemble</button>


<!-- Result text box -->

<textarea id="resultTextBox" rows="10" cols="40" readonly></textarea>
